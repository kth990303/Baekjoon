#2805 나무자르기 Silver III

#include <iostream>
using namespace std;
typedef long long ll;
const int INF=2000000007;
int N, M, a[1000001];
int main(){
	cin.tie(0);
	cout.tie(0);
	ios::sync_with_stdio(false);
	
	cin>>N>>M;
	for(int i=1;i<=N;i++){
		cin>>a[i];
	}
	int s=0, e=INF;
	while(s<=e){
		int mid=(s+e)/2;
		ll sum=0;
		for(int i=1;i<=N;i++){
			if(a[i]>mid)
				sum+=a[i]-mid;
		}
		if(sum>=M)
			s=mid+1;
		else
			e=mid-1;
	}
	cout<<s-1;
}
