#include <iostream>
using namespace std;
pair<int, int> p[46];
int main() {
	cin.tie(0);
	cout.tie(0);
	ios::sync_with_stdio(false);
	
	int k;
	cin>>k;
	p[0]={1,0};
	for(int i=1;i<=k;i++){
		p[i].first=p[i-1].second;
		p[i].second=p[i-1].first+p[i-1].second;
	}
	cout<<p[k].first<<" "<<p[k].second<<"\n";
	return 0;
}
