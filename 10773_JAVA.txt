package baekjoon;
import java.util.Scanner;
import java.util.Stack;
public class Num10773 {

	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		Stack s=new Stack();
		int sum=0;
		int k=sc.nextInt();
		for(int i=0;i<k;i++) {
			int n=sc.nextInt();
			if(n==0)
				s.pop();
			else {
				s.push(n);
			}
		}
		while(!s.empty()) {
			sum+=(int)s.peek();
			s.pop();
		}
		System.out.println(sum);
	}

}
