#include <iostream>
#include <algorithm>
using namespace std;
int getAns(int* arr, int n, int m) {
	int sum = 0, max = 0;
	for (int i = n - 1; i >= 0; i--) {
		sum += *(arr + i);
		for (int j = i - 1; j >= 0; j--) {
			sum += *(arr + j);
			for (int k = j - 1; k >= 0; k--) {
				sum += *(arr + k);
				if (sum <= m) {
					if (max < sum)
						max = sum;
					sum -= *(arr + k);
				}
				else
					sum -= *(arr + k);
			}
			sum -= *(arr + j);
		}
		sum -= *(arr + i);
	}
	return max;
 }
int main() {
	int n, m;
	cin >> n >> m;
	int* arr = new int[n];
	for (size_t i = 0; i < n; i++) {
		int num;
		cin >> num;
		*(arr + i) = num;
	}
	sort(arr, arr + n);
	cout<<getAns(arr, n, m);
}
